const revObserver=new IntersectionObserver((entries)=>{
  entries.forEach((e,i)=>{
    if(e.isIntersecting){
      // slight stagger for siblings
      setTimeout(()=>e.target.classList.add('in'),(i%6)*70);
      revObserver.unobserve(e.target);
    }
  });
},{threshold:.12});
document.querySelectorAll('.reveal').forEach(el=>revObserver.observe(el));

/* ---------- Animated counters ---------- */
const countObserver=new IntersectionObserver((entries)=>{
  entries.forEach(e=>{
    if(!e.isIntersecting)return;
    const el=e.target,target=+el.dataset.count,suffix=el.dataset.suffix||'';
    let cur=0;const step=Math.max(1,Math.floor(target/60));
    const tick=()=>{
      cur+=step;
      if(cur>=target){el.textContent=target+suffix;}
      else{el.textContent=cur+suffix;requestAnimationFrame(tick);}
    };
    tick();
    countObserver.unobserve(el);
  });
},{threshold:.5});
document.querySelectorAll('.num[data-count]').forEach(el=>countObserver.observe(el));

(function(){
  const canvas=document.getElementById('particles'),ctx=canvas.getContext('2d');
  let w,h,particles=[];
  const colors=['#00C6FF','#7B2CFF','#FF8A00'];
  function resize(){w=canvas.width=innerWidth;h=canvas.height=innerHeight;}
  function init(){
    const count=Math.min(70,Math.floor(innerWidth/22));
    particles=[];
    for(let i=0;i<count;i++){
      particles.push({
        x:Math.random()*w,y:Math.random()*h,
        r:Math.random()*2+.6,
        dx:(Math.random()-.5)*.4,dy:(Math.random()-.5)*.4,
        c:colors[Math.floor(Math.random()*colors.length)],
        a:Math.random()*.5+.2
      });
    }
  }
  function draw(){
    ctx.clearRect(0,0,w,h);
    particles.forEach(p=>{
      p.x+=p.dx;p.y+=p.dy;
      if(p.x<0||p.x>w)p.dx*=-1;
      if(p.y<0||p.y>h)p.dy*=-1;
      ctx.beginPath();
      ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
      ctx.fillStyle=p.c;ctx.globalAlpha=p.a;
      ctx.shadowBlur=8;ctx.shadowColor=p.c;
      ctx.fill();
    });
    ctx.globalAlpha=1;
    requestAnimationFrame(draw);
  }
  resize();init();draw();
  window.addEventListener('resize',()=>{resize();init();},{passive:true});
})();
